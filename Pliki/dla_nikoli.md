$$\begin{gathered}
zał:\\
0 \lt \alpha \lt \frac{\pi}{2}\\
\cos{\alpha}=\frac{\sqrt{7}}{4}\\\\\\\
\sin^2{\alpha}+\cos^2{\alpha}=1\\
\sin^2{\alpha}+\frac{7}{16}=1\\
\sin^2{\alpha}=1-\frac{7}{16}=\frac{9}{16}\\
\sin{\alpha}=\frac{3}{4} \vee \sin{\alpha}=-\frac{3}{4}\\\\
\text{A ponieważ}\\
0 \lt \alpha \lt \frac{\pi}{2}\\\\
\sin{\alpha}=\frac{3}{4}\\
\downarrow\\
\tan{\alpha}=\frac{\sin{\alpha}}{\cos{\alpha}}=\frac{3\sqrt{7}}{7}\\
\cot=\frac{1}{\tan}=\frac{\sqrt{7}}{3}
\end{gathered}$$
